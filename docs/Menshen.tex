> -out rsa_private_key.key

  \captionof{figure}{Générer la clé publique}
  \begin{minted}[linenos=true]{bash}
$ openssl pkey -in rsa_private_key.key \
-pubout > rsa_pubkey.pem
  \end{minted}
\end{minipage}
